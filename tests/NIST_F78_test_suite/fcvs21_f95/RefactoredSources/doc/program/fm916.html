!

           ivtnum = 1

        inquire(unit=iovi, exist=avb, opened=bvb, number=jvi,                     access=b10vk, direct=d10vk, recl=kvi, nextrec=lvi,                form=e11vk, formatted=f10vk, blank=h10vk, err=20014,              iostat=nvi)
