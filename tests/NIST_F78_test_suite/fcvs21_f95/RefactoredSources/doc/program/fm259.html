! Z01  ZVERS  = 'VERSION OF THE COMPILER VALIDATION SYSTEM'

      i02 = 06                                                          
!                                                                        
! X010   REPLACED BY FEXEC X-010 CONTROL CARD (CARD-READER UNIT NUMBER).

! BE** ********************** BBCINITB **********************************
      nuvi = i02                                                        
      ivtotl = 3                                                        
      zprog = 'FM259'                                                   
! BB** ********************** BBCHED0A **********************************
! ****

30100 format(" ", / " BLKIF2 - (301) BLOCK IF" //                              "  WITH OTHER CONTROL CONSTRUCTS (I)" //                          "  SUBSET REF.  11.1-11.3, 11.6-11.10" )

! BE** ********************** BBCHED0B **********************************
! *****                                                                  
           write (nuvi,70000)                                           
! T001*  TEST 1                                  IF (E) THEN .*. ENDIF   
           ivtnum = 1                                                   
           ivinsp = ivinsp + 1                                          
           write (nuvi, 80004) ivtnum                                   
        kvi = 7                                                         
        if (kvi  ==  7) then                                            
                kvi = 8                                                 
                if (kvi  >=  8) goto 0012                               
                kvi = 9                                                 
        endif                                                           
 0012   lvi = 8 - kvi                                                   
        write(nuvi,70010) lvi                                           
! T002*  TEST 2                          IF (E) THEN .*. ELSE .*. ENDIF  
           ivtnum = 2                                                   
           ivinsp = ivinsp + 1                                          
           write (nuvi, 80004) ivtnum                                   
        kvi = 0                                                         
        lvi = 1                                                         
        if (lvi  ==  7) then                                            
                kvi = 8                                                 
                if (lvi  ==  1) goto 0026                               
                kvi = 9                                                 
        else                                                            
                goto (0023, 0024, 0022), lvi                            
 0022           kvi = 1                                                 
                goto 0025                                               
 0023           kvi = 2                                                 
                goto 0025                                               
 0024           kvi = 3                                                 
 0025           continue                                                
        endif                                                           
 0026   lvi = 2 - kvi                                                   
        write(nuvi,70010) lvi                                           
! T003*  TEST 3                          DO ..........                   
           ivtnum = 3                                                   
           ivinsp = ivinsp + 1                                          
           write (nuvi, 80004) ivtnum                                   
        mvi = 0                                                         
        assign 0034 to nvi                                              
        lvi = 0                                                         
        jvi = 1                                                         
        do ivi = 1,4                                               
        if (ivi  ==  1) then                                            
                do kvi = 1,ivi                                     
                        lvi = lvi + 1                                   
                         end do
        elseif (ivi .eq. 2) then                                        
                lvi = 6                                                 
                if (.false.) goto 0036                                    !Break
                lvi = 2                                                 
        elseif (ivi .eq. 3) then                                        
                if (mvi  /=  0) then                                    
                        lvi = 7                                         
                else                                                    
                        lvi = 3                                         
                endif                                                   
        else                                                            
                goto nvi, (0033, 0034)                                  
 0033           lvi = 5                                                 
                goto 0035                                                 !Break
 0034           lvi = 4                                                 
                assign 0033 to nvi                                      
 0035           continue                                                
        endif                                                           
        lvi = lvi - jvi                                                 
 0036   write(nuvi,70010) lvi                                           
        jvi = jvi + 1                                                   
  end do
! *****                                                                  
! BB** ********************** BBCSUM0  **********************************

      write (i02, 90014)                                                
      write (i02, 90004)                                                
      write (i02, 90020) ivpass
