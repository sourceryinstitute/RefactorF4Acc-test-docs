! X020    THIS CARD IS REPLACED BY CONTENTS OF FEXEC X-020 CONTROL CARD.

 1040 continue

11040 ivpass = ivpass + 1

 1051 continue

!                                                                        
!       ****  TEST  105  ****                                            
!      TEST 105  -  TEST OF EMBEDDED BLANKS IN A REAL TYPE STATEMENT.

      if (iczero) 31050, 1050, 31050                                    
 1050 continue
